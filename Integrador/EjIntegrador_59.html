// En el archivo main.js document.addEventListener('DOMContentLoaded', function() { const dynamicDataContainer = document.getElementById('dynamic-data'); // Utiliza una API para obtener datos en tiempo real (por ejemplo, fetch()) fetch('https://api.deforestationdata.com/') .then(response => response.json()) .then(data => { // Procesa y muestra los datos dinámicos en la página dynamicDataContainer.innerHTML = `

Áreas Deforestadas en Tiempo Real

`; data.forEach(area => { dynamicDataContainer.innerHTML += `

${area.location}: ${area.amount} acres

`; }); }) .catch(error => { console.error('Error al obtener datos dinámicos:', error); dynamicDataContainer.innerHTML = '

Error al cargar datos dinámicos.

'; }); });
